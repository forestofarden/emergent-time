  def run name
    warn name
    Dir["**/*.rb"].sort.each do |path|
      next if path =~ /gemspec.rb/ # HACK
      next if data[name][path] == true
      process path, name
    end

    if self.data[name].values.all? { |v| v == true } then
      warn "  ALL GOOD!"
      self.data[name] = true
      self.dirty = true
    end
  end
