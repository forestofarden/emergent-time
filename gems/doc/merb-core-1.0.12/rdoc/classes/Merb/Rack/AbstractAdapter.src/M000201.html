      def self.spawn_worker(port)
        worker_pid = Kernel.fork
        start_at_port(port, @opts) unless worker_pid

        # If we have a worker_pid, we're in the parent.
        throw(:new_worker) unless worker_pid

        @pids[port] = worker_pid
        $WORKERS = @pids.values
      end
