      def alive?(port)
        pidfile = pid_file(port)
        pid     = pid_in_file(pidfile)
        Process.kill(0, pid)
        true
      rescue Errno::ESRCH, Errno::ENOENT
        false
      rescue Errno::EACCES => e
        Merb.fatal!("You don't have access to the PID file at #{pidfile}: #{e.message}")
      end
