      def bootup
        Merb.trap("TERM") { shutdown }

        Merb.logger.warn! "Running bootloaders..." if Merb::Config[:verbose]
        BootLoader.run
        Merb.logger.warn! "Starting Rack adapter..." if Merb::Config[:verbose]
        Merb.adapter.start(Merb::Config.to_hash)
      end
