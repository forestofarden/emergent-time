        def default_shortname
          self.email =~ /^(.*)\@/
          prefix = $1 || "#{first_name}#{last_name}"
          prefix = prefix.downcase.gsub(/\W+/, '')
          suffix = nil

          while self.class.first(:shortname => "#{prefix}#{suffix}") do
             suffix ||= 0
             suffix += 1
          end
          "#{prefix}#{suffix}"
        end
