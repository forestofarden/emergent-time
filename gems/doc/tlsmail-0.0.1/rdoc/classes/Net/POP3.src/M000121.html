    def initialize(addr, port = nil, isapop = false)
      @address = addr
      @use_ssl = POP3.use_ssl?
      @port = port || (POP3.use_ssl? ? POP3.default_pop3s_port : POP3.default_pop3_port)
      @apop = isapop
      @certs = POP3.certs
      @verify = POP3.verify
      
      @command = nil
      @socket = nil
      @started = false
      @open_timeout = 30
      @read_timeout = 60
      @debug_output = nil

      @mails = nil
      @n_mails = nil
      @n_bytes = nil
    end
