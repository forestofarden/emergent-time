      def config
        @config ||= begin
          empty_hash = Hash.new { |h,k| h[k] = {} }
          if File.exists?(Merb.root / "config" / "slices.yml")
            require "yaml"
            YAML.load(File.read(Merb.root / "config" / "slices.yml")) || empty_hash
          else
            empty_hash
          end
        end
      end
