      def manifest(type = :root)
        files = if type == :root
          Dir.glob(self.root / "**/*")
        elsif slice_paths.key?(type)
          glob = ((type == :view) ? view_templates_glob : glob_for(type) || "**/*")
          Dir.glob(dir_for(type) / glob)
        else 
          []
        end
        files.map { |source| [source, source.relative_path_from(root)] }
      end
