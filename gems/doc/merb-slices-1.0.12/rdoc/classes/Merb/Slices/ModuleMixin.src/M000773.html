      def push_app_path(type, path, file_glob = "**/*.rb")
        enforce!(type => Symbol)
        if type == :public && standalone? && $SLICE_MODULE
          path.gsub!(/\/slices\/#{self.identifier}$/, '')
        end
        app_paths[type] = [path, file_glob]
      end
