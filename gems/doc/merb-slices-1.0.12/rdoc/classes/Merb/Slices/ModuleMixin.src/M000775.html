      def loadable_files
        app_components.inject([]) do |paths, type|
          paths += Dir[dir_for(type) / '**/*.rb'] if slice_paths.key?(type)
          paths += Dir[app_dir_for(type) / '**/*.rb'] if app_paths.key?(type)
          paths
        end        
      end
